
planet_jobs = {
	parent = planets

	triggered_produces_modifier = {
		key = rsrchcap_purge
		modifier_types = {
			add
		}
		trigger = {
			OR = {
				has_job = purge
				has_job = event_purge
			}
			exists = owner
			owner = { has_valid_civic = civic_hive_devouring_swarm }
		}
	}
}
