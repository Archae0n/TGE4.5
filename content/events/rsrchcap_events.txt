	id = rsrchcap.2

	is_triggered_only = yes
	hide_window = yes
	fire_only_once = yes

	trigger = {
		event_target:global_event_country = {
			OR = {
				AND = {
					event_target:rsrchcap_compare_country = { is_same_value = event_target:rsrchcap_compare_country_t1 }
					check_variable = {
						which = rsrchcap_average_tech_ratio
						value >= 1.5
					}
				}
				check_variable = {
					which = rsrchcap_late_early_game_start
					value <= 0
				}
			}
		}
	}

	immediate = {
		clear_global_event_target = rsrchcap_compare_country

			save_global_event_target_as = rsrchcap_compare_country
			country_event = { id = rsrchcap.11 }
		}
	}
}

# Calculate the tech cost
country_event = {
	id = rsrchcap.20

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		is_country_type = rsrchcap_tech_cost_test_country
		NOT = { has_global_flag = rsrchcap_tech_cost_check }
	}

	immediate = {
		change_variable = {
			which = rsrchcap_months_passed
			value = 1
		}
		if = {
			limit = {
				OR = {
					is_researching_technology = rsrchcap_tech_cost_test
					has_technology = rsrchcap_tech_cost_test
				}
			}
			if = {
				limit = {
					has_country_flag = rsrchcap_tech_cost_not_researching
				}
				change_variable = {
					which = rsrchcap_tech_cost
					value = -0.25
				}
				remove_country_flag = rsrchcap_tech_cost_not_researching
			}
			change_variable = {
				which = rsrchcap_tech_cost
				value = 0.25
			}
			log = "Researching Tech ([this.rsrchcap_tech_cost])"

			if = {
				limit = {
					has_technology = rsrchcap_tech_cost_test
				}
				event_target:global_event_country = {
					set_variable = {
						which = rsrchcap_tech_cost
						value = root
					}
					set_variable = {
						which = rsrchcap_months_passed
						value = root
					}
					if = {
						limit = {
							has_global_flag = rsrchcap_scale_tech_cost
						}
						multiply_variable = {
							which = rsrchcap_late_early_game_start
							value = rsrchcap_tech_cost
						}
						multiply_variable = {
							which = rsrchcap_mid_game_start
							value = rsrchcap_tech_cost
						}
						multiply_variable = {
							which = rsrchcap_late_mid_game_start
							value = rsrchcap_tech_cost
						}
						multiply_variable = {
							which = rsrchcap_end_game_start
							value = rsrchcap_tech_cost
						}
						multiply_variable = {
							which = rsrchcap_late_end_game_start
							value = rsrchcap_tech_cost
						}
					}
					divide_variable = {
						which = rsrchcap_months_passed
						value = 12
					}
					divide_variable = {
						which = rsrchcap_months_passed
						value = 100000
					}
					multiply_variable = {
						which = rsrchcap_months_passed
						value = 100000
					}
					subtract_variable = {
						which = rsrchcap_late_early_game_start
						value = rsrchcap_months_passed
					}
					subtract_variable = {
						which = rsrchcap_mid_game_start
						value = rsrchcap_months_passed
					}
					subtract_variable = {
						which = rsrchcap_late_mid_game_start
						value = rsrchcap_months_passed
					}
					subtract_variable = {
						which = rsrchcap_end_game_start
						value = rsrchcap_months_passed
					}
					subtract_variable = {
						which = rsrchcap_late_end_game_start
						value = rsrchcap_months_passed
					}
					change_variable = {
						which = rsrchcap_late_early_game_start
						value = 0.5
					}
					change_variable = {
						which = rsrchcap_mid_game_start
						value = 0.5
					}
					change_variable = {
						which = rsrchcap_late_mid_game_start
						value = 0.5
					}
					change_variable = {
						which = rsrchcap_end_game_start
						value = 0.5
					}
					change_variable = {
						which = rsrchcap_late_end_game_start
						value = 0.5
					}
					divide_variable = {
						which = rsrchcap_late_early_game_start
						value = 10000
					}
					divide_variable = {
						which = rsrchcap_mid_game_start
						value = 10000
					}
					divide_variable = {
						which = rsrchcap_late_mid_game_start
						value = 10000
					}
					divide_variable = {
						which = rsrchcap_end_game_start
						value = 10000
					}
					divide_variable = {
						which = rsrchcap_late_end_game_start
						value = 10000
					}
					multiply_variable = {
						which = rsrchcap_late_early_game_start
						value = 10000
					}
					multiply_variable = {
						which = rsrchcap_mid_game_start
						value = 10000
					}
					multiply_variable = {
						which = rsrchcap_late_mid_game_start
						value = 10000
					}
					multiply_variable = {
						which = rsrchcap_end_game_start
						value = 10000
					}
					multiply_variable = {
						which = rsrchcap_late_end_game_start
						value = 10000
					}
					set_global_flag = rsrchcap_tech_cost_check
					log = "Tech Cost Found: [This.rsrchcap_tech_cost]"
					break = yes
				}
			}
		}
		else = {
			set_country_flag = rsrchcap_tech_cost_not_researching
		}
		country_event = { id = rsrchcap.20 days = 30 }
	}
}
