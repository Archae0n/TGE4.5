
namespace = SGM_SL_event

#Control SL enable/disable
#Triggered on_monthly_pulse.
event = {
	id = SGM_SL_event.90
	hide_window = yes
	is_triggered_only = yes
		
	immediate = {
	
		every_country={
			if = {
                limit={
                    is_ai=yes
                    is_at_war=yes
                    NOT={has_global_flag=SL_DISABLE_FOR_AI}
                    OR={	
                        is_country_type = default
                        is_country_type = fallen_empire
                        is_country_type = awakened_fallen_empire
                    }
				}
				set_country_flag=SL_ENABLE
			}
			if={
                limit={
                    is_ai=yes
                    has_global_flag=SL_DISABLE_FOR_AI
				}
				remove_country_flag=SL_ENABLE
			}
			if={
                limit={is_ai=no}
				set_country_flag=SL_ENABLE
			}
		}
	}
}

#Remove ships that were removed from the SL fleets
#Triggered on_yearly_pulse.
event = {
	id = SGM_SL_event.91
	hide_window = yes
	is_triggered_only = yes
		
	immediate = {	
		every_country={
			limit={is_country_type = SGM_supply_corps}				
			every_owned_ship={
				limit={
					has_ship_flag=SL_ship
					fleet={NOT={has_fleet_flag=SL_fleet}}
				}
				log="CleanUp: Destroying ship [Prev.GetName]-[This.GetName]"
				reduce_hp=999999 
			}			
		}
	}
}

#Triggered on_game_start
event = {
    id = SGM_SL_event.921
    hide_window = yes
    is_triggered_only = yes
    
    immediate = {
        random_country={
            limit = { is_ai = no }
            country_event = {
                id = SGM_SL_event.922
                days = 1
            }
        }
    }
}

#Triggered on_game_start + 1 day
country_event = {
	id = SGM_SL_event.922
	hide_window = yes
	is_triggered_only = yes
		
	immediate = {
        every_ship = { # convert advanced start scout ships to corvette
            limit = {
                is_ship_size = SGM_scout
            }
            fleet = {
				create_ship = {
                    name = random
                    random_existing_design = corvette
                }
            }
            reduce_hp = 999999
        }
		every_country={
            limit = {
                OR={
                    is_country_type = default
                    is_country_type = fallen_empire
                    is_country_type = awakened_fallen_empire
                    is_country_type = awakened_fallen_empire
                    is_country_type = SGM_supply_corps
                }
            }
            # log="Launched supply events for [This.GetName]"
			country_event={id=SGM_SL_event.94} #Check connection overlord <-> supply corps			 
			country_event={id=SGM_SL_event.95} #Clean up left over supply corps
			country_event={id=SGM_SL_event.93} #check for new countries
		}
	}
}

		OR={
			is_country_type = default
			is_country_type = fallen_empire

	}
		
	immediate = {	
		
		#check connection of supply corps
		SGM_country2corps = yes #SL corps with matching id saved in event_target:SL_corps
		
		if={limit={exists=event_target:SL_corps}
			#country -> corps ok
			
			event_target:SL_corps={
				SGM_corps2country = yes #country with matching id saved in event_target:SL_country
			}
			
			if={limit={event_target:SL_country={is_same_value = root}}
				#corps -> country ok
				
				event_target:SL_corps={
					#check vassal status
					if={limit={							
							NOT={
								AND = {									
									is_subject_type = SGM_SL_vassal
									overlord = { is_same_value = root}
								}
							}
						}
						log="WARNING: Resetting SL corps [This.GetName] to be vassal of country [Prev.GetName]!"
						set_subject_of = {
							who = root
							subject_type = SGM_SL_vassal
						}					
					}
					#check if any planets have been claimed and give them to the overlord
					every_owned_planet={
						log="Warning: Changed planet overship of planet [This.GetName] because it belonged to a supply corps"
						set_owner = event_target:SL_country
					}
				}
				log="Success2: Supply corps status of [This.GetName] is ok"
			}
			else={
				#corps -> country not ok
				event_target:SL_corps={destroy_country=yes}
				country_event={id=SGM_SL_event.83} #reset country and create new supply corps
			}
        }
		else={
			#country -> corps not ok
			country_event={id=SGM_SL_event.83} #reset country and create new supply corps
		}
	}
}

# From = system
# FromFrom = country (previous owner)
# This = country (new owner)
# on_system_gained
country_event = {
    id = SGM_SL_event.101
    hide_window = yes
    is_triggered_only = yes

	trigger={
        is_country_type = SGM_supply_corps
	}
	
	immediate = {
		# If this supply corp has an overlord, give him the system
        if = {
            limit = {
                exists = overlord
            }
            SGM_corps2country = yes # overlord in event_target:SL_country
            from = {
                starbase = {
                    set_owner = event_target:SL_country
                }
            }
        }
        # If this supply corp doesn't have an overlord (for some reason)
        else = {
            # If the previous owner still exists
            if = {
                limit = {
                    exists = FromFrom
                }
                from = {
                    set_owner = FromFrom
                }
            }
            else = {
                starbase = { fleet = { destroy_fleet = this } }
            }
        }
    }
}
